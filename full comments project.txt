import java.util.InputMismatchException; // Importing InputMismatchException class for handling input errors
import java.util.Scanner; // Importing Scanner class for user input

public class AdvancedTemperatureConverter {

    // Method to convert Celsius to Fahrenheit
    public static double celsiusToFahrenheit(double celsius) {
        return (celsius * 9 / 5) + 32;
    }

    // Method to convert Fahrenheit to Celsius
    public static double fahrenheitToCelsius(double fahrenheit) {
        return (fahrenheit - 32) * 5 / 9;
    }

    // Method to convert Celsius to Kelvin
    public static double celsiusToKelvin(double celsius) {
        return celsius + 273.15;
    }

    // Method to convert Kelvin to Celsius
    public static double kelvinToCelsius(double kelvin) {
        return kelvin - 273.15;
    }

    // Method to convert Fahrenheit to Kelvin
    public static double fahrenheitToKelvin(double fahrenheit) {
        return (fahrenheit - 32) * 5 / 9 + 273.15;
    }

    // Method to convert Kelvin to Fahrenheit
    public static double kelvinToFahrenheit(double kelvin) {
        return (kelvin - 273.15) * 9 / 5 + 32;
    }

    // Method to get a valid double input from the user
    private static double getValidDoubleInput(Scanner scanner) {
        double value;//Variable to store the valid double value
        while (true) //Infinite loop until a valid input is obtained
        {
            try {
                value = scanner.nextDouble(); // Reading a double value from the user
                break; // Break the loop if input is valid
            } catch (InputMismatchException e) {
                System.out.print("Invalid input. Please enter a valid number: "); // Prompt the user for valid input
                scanner.next(); // Clear the invalid input from the scanner
            }
        }
        return value; // Return the valid double value
    }

    // Method to get a valid integer input from the user
    private static int getValidIntInput(Scanner scanner, int min, int max) {
        int choice;
        while (true) {
            try {
                choice = scanner.nextInt(); // Reading an integer value from the user
                if (choice >= min && choice <= max) { // Checking if the input is within the specified range
                    break; // Break the loop if input is valid
                } else {
System.out.print("Invalid choice. Please choose a valid option (" + min + "-" + max + "): "); // Prompt the user for valid input
                }
            } catch (InputMismatchException e) {
     System.out.print("Invalid input. Please enter a valid option (" + min + "-" + max + "): "); // Prompt the user for valid input
                scanner.next(); // Clear the invalid input from the scanner
            }
        }
        return choice; // Return the valid integer value
    }

    // Main method
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in); // Creating a Scanner object to read user input

        while (true) {
            // Displaying the menu options
            System.out.println("\nTemperature Converter Menu:");
            System.out.println("1. Celsius to Fahrenheit");
            System.out.println("2. Fahrenheit to Celsius");
            System.out.println("3. Celsius to Kelvin");
            System.out.println("4. Kelvin to Celsius");
            System.out.println("5. Fahrenheit to Kelvin");
            System.out.println("6. Kelvin to Fahrenheit");
            System.out.println("7. Exit");
            System.out.print("Choose an option (1-7): ");

            // Getting a valid integer input for menu choice
            int choice = getValidIntInput(scanner, 1, 7);

            // Exiting the program if the user chooses option 7
            if (choice == 7) {
                System.out.println("Exiting the program. Goodbye!");
                break;
            }

            // Prompting the user for the temperature to convert
            System.out.print("Enter the temperature to convert: ");
            double inputTemperature = getValidDoubleInput(scanner); // Getting a valid double input for temperature
            double resultTemperature = 0;

            // Performing the selected conversion and displaying the result
            switch (choice) {
                case 1://if chose is 1//
                    resultTemperature = celsiusToFahrenheit(inputTemperature);//convert  celsiusToFahrenheit//
                    System.out.printf("%.2f Celsius is %.2f Fahrenheit.%n", inputTemperature, resultTemperature);//print the result//
                    break;// exit the swith statement//
                case 2://if chose is 2//
                    resultTemperature = fahrenheitToCelsius(inputTemperature);//convert fahrenheitToCelsius//
                    System.out.printf("%.2f Fahrenheit is %.2f Celsius.%n", inputTemperature, resultTemperature);//print the result//
                    break;// exit the swith statement//
                case 3://if chose is 3//
                    resultTemperature = celsiusToKelvin(inputTemperature);// convert celsiusToKelvin//
                    System.out.printf("%.2f Celsius is %.2f Kelvin.%n", inputTemperature, resultTemperature);//print the result//
                    break;// exit the swith statement//
                case 4://if chose is 4//
                    resultTemperature = kelvinToCelsius(inputTemperature);//convert kelvinToCelsius//
                    System.out.printf("%.2f Kelvin is %.2f Celsius.%n", inputTemperature, resultTemperature);//print the result//
                    break;// exit the swith statement//
                case 5://if chose is 5//
                    resultTemperature = fahrenheitToKelvin(inputTemperature);//convert fahrenheitToKelvin//
                    System.out.printf("%.2f Fahrenheit is %.2f Kelvin.%n", inputTemperature, resultTemperature);//print the result//
                    break;// exit the swith statement//
                case 6:// if chose is 6//
                    resultTemperature = kelvinToFahrenheit(inputTemperature);// convert kelvinToFahrenheit//
                    System.out.printf("%.2f Kelvin is %.2f Fahrenheit.%n", inputTemperature, resultTemperature);//print the result//
                    break;// exit the swith statement//
                default:
                    System.out.println("Invalid choice. Please choose a valid option.");
                    break;// exit the swith statement//
            }
        }
        scanner.close(); // Closing the Scanner object to release resources
    }
}
